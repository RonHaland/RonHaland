Import-Module posh-git


## Git functions
function commit { git commit -m $args }
function amend { git commit --amend --no-edit }
function push { git push }
function pull { git pull }
function fetch { git fetch }
function fetch-rebase { git fetch && git rebase $args }
function delete-branches { & $env:HOME/.config/powershell/delete-git-branches.ps1 -Pattern $args[0] }


## Git aliases
Set-Alias -Name gcom -Value commit
Set-Alias -Name gamend -Value amend
Set-Alias -Name gpus -Value push
Set-Alias -Name gpul -Value pull
Set-Alias -Name gfr -Value fetch-rebase
Set-Alias -Name gdel -Value delete-branches

## GPG
$env:GPG_TTY = (tty)

## Present markdown
function present-markdown { node $env:HOME/dev/RonHaland/javascript/scripts/present-markdown.mjs $args }
Set-Alias -Name present -Value present-markdown
